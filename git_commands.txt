git bash commands

clone:	new repository..
add:	add new directory or file(s)
commit:	notify change on dir/file(s)
push:	upload changes
pull:	download changes

// ablauf
lokal aktualisieren:
	1) git pull
	
änderungen aktualisieren:
	1) git commit ..
	2) git push
	
dateien hinzufügen:
	1) git add ..
	2) git commit
	3) git push
